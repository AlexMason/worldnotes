```typescript
function insertTextAtSelection(text: string): void {
  const sel = window.getSelection()
  if (!sel || !sel.rangeCount) return

  const range = sel.getRangeAt(0)
  range.deleteContents()

  const node = document.createTextNode(text)
  range.insertNode(node)
  range.setStart(node, text.length)
  range.collapse(true)

  sel.removeAllRanges()
  sel.addRange(range)

  dom.editorDiv.dispatchEvent(new Event('input', { bubbles: true }))
}
```

```typescript
function mountEditor(
  container: HTMLElement,
  plugins: Plugin[],
  storage: StorageAdapter,
  options: EditorOptions,
): EditorInstance {
  // 1. Initialize state
  const state = createEditorState(storage, options)

  // 2. Build DOM
  const dom = createEditorDOM(container)

  // 3. Create navigation (needs state, storage, dom)
  //    Navigation does not need render at construction time — it receives it via setRenderAPI
  const navigation = createEditorNavigation(state, storage, dom, options)

  // 4. Create render pipeline (needs dom, plugins, state, navigation callbacks)
  const renderOptions: EditorRenderOptions = {
    navigateFn: (page: string) => { navigation.navigateToPage(page) },
    onBreadcrumbNavigate: (page: string) => { navigation.loadPage(page) },
    onTrailChange: options.onTrailChange,
  }
  const render = createEditorRender(dom, plugins, state, renderOptions)

  // 5. Wire render into navigation (so loadPage can call render/renderBreadcrumb)
  navigation.setRenderAPI(render)

  // 6. Create lifecycle (needs everything, returns EditorInstance)
  const lifecycle = createEditorLifecycle(dom, plugins, state, render, navigation, storage, options)

  // 7. Mount and return live EditorInstance
  return lifecycle.mount()
}
```

**Construction order rationale** (per DAG):
1. `createEditorState` — root, no editor-* deps
2. `createEditorDOM` — pure DOM, no editor-* deps
3. `createEditorNavigation` — needs state + storage + dom
4. `createEditorRender` — needs dom + plugins + state + navigate callbacks
5. `navigation.setRenderAPI(render)` — wires two-phase construction
6. `createEditorLifecycle` — needs everything, assembles EditorInstance
7. `lifecycle.mount()` — attaches event listeners, calls loadPage
